# emerge --ask dev-db/postgresql

# passwd postgres
# mkdir /run/postgresql

# mkdir /usr/local/pgsql

$ /usr/lib64/postgresql-9.4/bin/initdb -E UTF-8 -D /usr/local/pgsql/data

#!/bin/bash
#
#	hello.sh
#
while true
do
	echo hello world >> /tmp/hello.log
	date >> /tmp/hello.log
	sleep 1
done
/etc/systemd/system/hello.service

[Unit]
Description = hello daemon

[Service]
ExecStart = /opt/hello.sh
Restart = always
Type = simple

[Install]
WantedBy = multi-user.target

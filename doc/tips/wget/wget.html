wget

インストール

# pacman -S wget

使用例
USER='scott'
PASSWORD='tiger'
SRC=ftp://192.168.18.8/httpdocs
#
wget -r -l 0 --user=$USER --password=$PASSWORD $SRC"/" >& /tmp/tmp01
mv 192.168.18.8/httpdocs .
#
rm -rf 192.168.18.8
DropBox からのダウンロード

wget --no-check-cert 'https://www.dropbox.com/s/71f8zyff72a52l9/words_idf.json?dl=0'

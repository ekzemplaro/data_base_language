samba サーバーの設定

インストール

# pacman -S samba

# pacman -S smbclient

/etc/samba/smb.conf の設定
# cp /etc/samba/smb.conf.default /etc/samba/smb.conf
	[tmp]
	   comment = Temporary file space
	   path = /tmp
	   read only = no
	   public = yes
ユーザーの設定
# pdbedit -a -u uchida

起動
# systemctl start nmbd

# systemctl start smbd

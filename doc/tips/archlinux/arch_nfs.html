nfs サーバーの設定

インストール

# pacman -S nfs-utils

/etc/exports の設定
	# /etc/exports: NFS file systems being exported.  See exports(5).
	/tmp    host_aa(async,rw,no_subtree_check,fsid=0) host_bb(async,rw,no_subtree_check,fsid=0)
	/home/uchida    host_aa(async,rw,no_subtree_check) host_bb(async,rw,no_subtree_check)
	/var/www   host_aa(async,rw,no_subtree_check) host_bb(async,rw,no_subtree_check)
	#
nfs の起動
# systemctl start nfs-server.service
自動的に起動させる。
# systemctl enable nfs-server.service
状態の表示
# systemctl status nfs-server.service
NFS クライアント
# systemctl start nfs-client.target

# systemctl start rpcbind.service
